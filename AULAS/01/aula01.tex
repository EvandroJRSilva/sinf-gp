\begin{frame}{Introdução}
	\begin{block}{O que é um projeto?}
		\justifying
		Um projeto consiste em um conjunto de tarefas que precisam ser completadas para se alcançar um objetivo predefinido.
	\end{block}
	\begin{itemize}
		\justifying
		\item Um projeto possui as seguintes características:
			\begin{enumerate}
				\justifying
				\item<2-> \textbf{Produz uma saída nova e tangível}: o principal objetivo de um projeto é criar algo novo.
				\item<3-> \textbf{É limitado pelo tempo}: um projeto tem sempre uma data de início e fim.
				\item<4-> \textbf{Existe fora das rotinas normais de um negócio}: um projeto é uma atividade distinta que ocorre além das operações normais e cotidianas de uma empresa.
				\item<5> \textbf{Possui limites}: um projeto é limitado por diversos fatores como tempo, custo, recursos e entregáveis.
			\end{enumerate}
	\end{itemize}
\end{frame}

\begin{frame}{Gerência de Projetos}

		\justifying
		\item<1-> A Gerência de Projetos se refere à forma de se \textbf{organizar}, \textbf{supervisionar} e \textbf{completar} a criação de um \alert<3->{\textbf{sistema de entrega de valor}}.
		\item<2> Um \textbf{gerenciamento bem sucedido} requer \textbf{conhecimento especí-fico}, \textbf{habilidades}, \textbf{ferramentas} e \textbf{técnicas} para executar eficazmente um projeto dentro de parâmetros, como prazo e orçamento.

\subsection{Criação de valor}

\begin{frame}{Criação de valor}
	\begin{itemize}
		\justifying
		\item Um sistema de entrega de valor define como os gerentes podem gerar benefícios para sua organização e para as \textbf{partes interessadas} (\textit{stakeholders}), utilizando uma governança que mede o sucesso não apenas por entregas, mas pelo valor final alcançado.
		\item<2-> Exemplos das formas com que os projetos produzem valor:
		\begin{itemize}
			\justifying
			\item<2> Criação de um novo produto, serviço ou resultado que atenda às necessidades de clientes ou usuários finais.
			\item<3> Criação de contribuições positivas, sociais ou ambientais.
			\item<4> Aprimoramento da eficiência, a produtividade, a eficácia ou a capacidade de resposta.
			\item<5> Viabilização das mudanças necessárias para facilitar a transição organizacional para seu futuro estado desejado.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Criação de valor}
	\centering
	\includegraphics[width=0.9\textwidth]{imagens/pmbok-2-1.png}
	\begin{itemize}
		\justifying
		\item Componentes, como portfólios, programas, projetos, produtos e ope-rações, podem ser usados individual e coletivamente para criar valor.
	\end{itemize}
\end{frame}

\begin{frame}{Criação de valor}
	\centering
	\includegraphics[width=0.7\textwidth]{imagens/pmbok-2-2.png}
	\begin{itemize}
		\small
		\justifying
		\item Um sistema de entrega de valor faz parte do \textbf{ambiente interno} da organi-zação e está sujeito a \textbf{políticas}, \textbf{procedimentos}, \textbf{metodologias}, \textit{\textbf{frame-works}}, \textbf{estruturas de governança} e assim por diante.
		\item Esse ambiente interno existe no âmbito do \textbf{ambiente externo} mais amplo, que inclui a \textbf{economia}, o \textbf{ambiente competitivo}, as \textbf{restrições legislativas}, etc.
	\end{itemize}
\end{frame}

\begin{frame}{Criação de valor}
	\begin{itemize}
		\justifying
		\item \textbf{Fatores internos} da organização podem surgir da organização em si, de um portfólio, programa, outro projeto ou uma combinação destes elementos. Entre os fatores incluem-se \textbf{artefatos}, \textbf{práticas} ou \textbf{conhecimento interno}. Exemplos:
			\begin{itemize}
				\justifying
				\item<2> \textbf{Ativos de processo}: podem incluir ferramentas, metodologias, abordagens, modelos, \textit{frameworks}, padrões, etc.
				\item<3> \textbf{Documentação de governança}: inclui políticas e processos.
				\item<4> \textbf{Ativos de dados}: podem incluir bancos de dados, bibliotecas de documentos, métricas, dados e artefatos de projetos anteriores.
				\item<5> \textbf{Cultura, estrutura e governança organizacionais}: incluem visão, missão, valores, crenças, normas culturais, estilo de liderança, relações de autoridade e hierarquia, estilo organizacional, ética e código de conduta.
			\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Criação de valor}
	\begin{itemize}
		\justifying
		\item Os \textbf{fatores externos} à organização podem ser neutros, ampliar ou restringir os resultados do projeto. Exemplos:
		\begin{itemize}
			\justifying
			\item<2> \textbf{Condições de mercado}: incluem concorrentes, participação no mercado, reconhecimento de marca, tendências tecnológicas e marcas registradas.
			\item<3> \textbf{Ambiente regulatório}: pode incluir leis nacionais e regionais, e regulamentos relativos à segurança, proteção de dados, conduta comercial, emprego, licenciamento e aquisições.
			\item<4> \textbf{Padrões do setor}: referem-se a produtos, produção, ambiente, qualidade e mão de obra.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Criação de valor}
	\centering
	\includegraphics[width=\textwidth]{imagens/pmbok-2-3.png}
	\begin{itemize}
		\justifying
		\item Um sistema de entrega de valor \textbf{funciona com mais eficácia} se as \textbf{informações e o \textit{feedback}} forem \textbf{compartilhados coerentemente} entre todos os componentes, mantendo o sistema alinhado com a estratégia e sintonizado com o ambiente.
	\end{itemize}
\end{frame}

\section{PMBOK}

\begin{frame}{PMBOK}
	\begin{itemize}
		\justifying
		\item O \alert<2>{PMBOK} (\textit{\textbf{P}roject \textbf{M}anagement \textbf{B}ody \textbf{o}f \textbf{K}nowledge} - Corpo de Conhecimento de Gerenciamento de Projeto) consiste em um \textbf{conjunto de terminologias e guias padronizados} para o \textbf{gerenciamento de projetos}.
		\begin{itemize}
			\justifying
			\item<2> É ``formalizado'' em um livro chamado \textbf{Guia PMBOK}. 
			\item<2> Mais especificamente o livro tem o seguinte título: Padrão de Gerenciamento de Projetos e Guia do Conhecimento em Gerenciamento de Projetos.
		\end{itemize}
		\item<3-> Quem organiza e publica o livro é o \href{https://www.pmi.org/}{\underline{\textbf{PMI}}} (\textit{Project Management Institute} - \alert<4>{Instituto de Gerenciamento de Projetos}).
		\begin{itemize}
			\justifying
			\item<4> É uma instituição internacional sem fins lucrativos e associação profissional de gerentes/gestores de projetos.
			\item<4> Já são mais de 680 mil associados.
			\item<4> O mais próximo de nós é o capítulo \href{https://pmice.org.br/}{\underline{PMI Ceará, Brasil}}.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{PMBOK}
	\begin{itemize}
		\justifying
		\item O livro \textbf{Guia PMBOK} é distribuído gratuitamente, mas é necessário criar uma conta no site. Também \alert<2>{está em constante atualização}.
		\begin{itemize}
			\justifying
			\item<2> A primeira edição é de 1996, e a sétima de 2021. Espera-se que no próximo ano (2026) saia a oitava edição.
			\item<3-> É separado em duas partes: (1) \alert<4>{\textbf{Padrão de Gerenciamento de Projetos}} e (2) \alert<5>{\textbf{Guia do Conhecimento em Gerenciamento de Projetos}}.
			\begin{itemize}
				\justifying
				\item<4> Descreve o padrão a ser seguido, independente de tamanho, setor, localização ou métodos de entrega.
				\item<5> Descreve a ligação entre o Padrão de Gerenciamento de Projetos e o Guia PMBOK.
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{PMBOK}
	\begin{itemize}
		\justifying
		\item \textbf{Padrão de Gerenciamento de Projetos} -- possui 3 seções:
		\begin{enumerate}
			\justifying
			\item<1> \textbf{Introdução}: descreve o propósito do padrão, define os termos e conceitos principais, e identifica o público do padrão.
			\item<2> \textbf{Um sistema para entrega de valor}: define como os gerentes de projetos podem criar valor para organizações e \textit{stakeholders}; um sistema de governança para medir o valor criado; uma lista de operações de suporte ao projeto; a influência de ambientes internos e externos, e a interconexão entre projetos, portfólios, programas e produtos.
			\item<3> \textbf{Princípios do gerenciamento de projetos}: foca em fornecer guias para os gerentes de projeto gerenciarem pessoas. Os membros da equipe precisam aderir a esses princípios para garantir os resultados almejados.
		\end{enumerate}
	\end{itemize}
\end{frame}

\begin{frame}{PMBOK}
	\begin{itemize}
		\justifying
		\item \textbf{Guia do Conhecimento em Gerenciamento de Projetos} -- possui 4 seções:
		\begin{enumerate}
			\justifying
			\item<1> \textbf{Introdução}: descreve a relação do Guia PMBOK com o Padrão de Gerenciamento de Projetos, as alterações aplicadas ao Guia PMBOK, e a relação com a PMIstandards+ (plataforma digital do PMI para padrões).
			\item<2> \textbf{Domínios de desempenho de projetos}: identifica e descreve oito domínios de desempenho de projetos que compõem um sistema integrado para viabilizar a entrega bem-sucedida do projeto e os resultados pretendidos.
			\item<3> \textbf{\textit{Tailoring}}: descreve o significado de \textit{tailoring} e apresenta uma visão geral do que pode ser classificado como \textit{tailoring} e como fazer o \textit{tailoring} em cada projeto.
			\item<4> \textbf{Modelos, métodos e artefatos}: apresenta uma descrição sucinta dos modelos, métodos e artefatos usados comumente.
		\end{enumerate}
	\end{itemize}
\end{frame}
