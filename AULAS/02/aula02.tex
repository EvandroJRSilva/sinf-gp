            \begin{itemize}

                \item<2> \textbf{Desafio}: equilibrar as habilidades técnicas, experiências e personalidades $\rightarrow$ grupo coeso.
                \item<3> ``O melhor tamanho para um grupo de engenharia de software é de quatro a seis membros e eles nunca devem ter mais de 12 membros'' (Sommerville).
            \end{itemize}
    \end{itemize}
\end{frame}

\subsection{Grupo Coeso}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Seus membros
            \begin{itemize}
                \justifying
                \item<1-> São \textbf{motivados pelo sucesso do grupo} e também, por seus pró-prios objetivos pessoais.
                \item<2-> São \textbf{leais} e \textbf{consideram o grupo mais importante} do que os indivíduos que o compõem.
                \item<3-> \textbf{Se identificam com os objetivos do grupo} e com os demais membros.
                \item<4> \textbf{Tentam proteger o grupo}, como uma entidade, de interferências externas.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Benefícios de um grupo coeso no desenvolvimento de software
            \begin{enumerate}
                \justifying
                \item<1> \textbf{O grupo pode estabelecer seus próprios padrões de qualidade}: Como esses padrões são estabelecidos por consenso, eles estão mais propensos a serem seguidos do que os padrões externos impostos ao grupo.
                \item<2> \textbf{Indivíduos aprendem e apoiam uns aos outros}: Os membros do grupo aprendem trabalhando juntos. As inibições causadas pela ignorância são minimizadas à medida que a aprendizagem mútua é incentivada.
            \end{enumerate}
    \end{itemize}
\end{frame}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Benefícios de um grupo coeso no desenvolvimento de software
        \begin{enumerate}
            \justifying
            \setcounter{enumi}{2}
            \item<1> \textbf{O conhecimento é compartilhado}: A continuidade pode ser mantida se um membro do grupo sair. Outros no grupo podem assumir tarefas criticas e assegurar que o projeto não seja indevidamente interrompido.
            \item<2> \textbf{A refatoração e a melhoria continua são incentivadas}: Os membros do grupo trabalham coletivamente para fornecer resultados de alta qualidade e corrigir problemas, independentemente dos indivíduos que criaram o projeto (design) ou programa originalmente.
        \end{enumerate}
    \end{itemize}
\end{frame}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Bons gerentes de projetos devem sempre tentar incentivar a coesão do grupo
            \begin{itemize}
                \justifying
                \item<1> Podem tentar estabelecer um senso de identidade do grupo ao nomeá-lo, estabelecer uma identidade e um território. [Time Macintosh vs. Restante da Apple -- Piratas do Vale do Silício].
                \item<2> Atividades de formação de grupo explícitas, como esportes e jogos [Top Gun Maverick], eventos sociais, inclusive envolvendo as famílias [Megadeth].
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Situações que atrapalham a formação de um grupo coeso
            \begin{itemize}
                \justifying
                \item Organização e bem estar dos funcionários em relação à empresa.
                \item O status do projeto: risco de ser cancelado, mudanças constantes, etc.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Grupo Coeso}
    \begin{itemize}
        \justifying
        \item Pontos principais que exigem atenção do gerente:
            \begin{itemize}
                \justifying
                \item Selecionar os membros.
                \item Organização do grupo.
                \item Comunicação.
            \end{itemize}
    \end{itemize}
\end{frame}

\section{Gerenciamento de riscos}

\setbeamertemplate{background canvas}{%
    \tikz[remember picture,overlay] \node[opacity=0.15,inner sep=0pt] at (current page.center){\includegraphics[width=\paperwidth,height=\paperheight]{imagens/backgrounds/risco.png}};
}

\begin{frame}{Gerenciamento de riscos}
    \begin{itemize}
        \justifying
        \item É a \textbf{atividade} de \textbf{prever} e \textbf{se preparar} para eventuais casualidades que interfiram no desenvolvimento do projeto.
        \item Podem ser categorizados de acordo com o que afetam e, também, de acordo com o seu tipo.
    \end{itemize}
\end{frame}

\begin{frame}{Gerenciamento de riscos}
    \begin{itemize}
        \justifying
        \item Riscos de acordo com o que afetam:
        \begin{itemize}
            \justifying
            \item<1> \textbf{Riscos de projeto}: Afetam o cronograma ou os recursos. Exemplo: perda de algum membro experiente da equipe.
            \item<2> \textbf{Riscos de produto}: Afetam a qualidade ou o desempenho do software que está sendo desenvolvido. Exemplo: mal funcionamento de alguma ferramenta, ou hardware.
            \item<3> \textbf{Riscos de negócio}: Afetam a empresa que está desenvolvendo ou comprando o software. Exemplo: uma empresa concorrente lança novos produtos antes dos seus (ao mesmo tempo que é um risco, pode ser uma oportunidade).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Gerenciamento de riscos}
    \begin{itemize}
        \justifying
        \item Riscos de acordo com seu tipo:
        \begin{itemize}
            \justifying
            \item<1-> \textbf{Estimativa}: errar a estimativa de recursos necessários.
            \item<2-> \textbf{Organizacionais}: problemas no ambiente de trabalho.
            \item<3-> \textbf{Pessoas}: problemas com pessoas durante o desenvolvimento.
            \item<4-> \textbf{Requisitos}: quando há mudanças nos requisitos.
            \item<5-> \textbf{Tecnologia}: problemas com hardware e software utilizados no desenvolvimento do projeto.
            \item<6> \textbf{Ferramentas}: problemas nas ferramentas utilizadas no desenvolvimento.
        \end{itemize}
    \end{itemize}
\end{frame}

\subsubsection{Identificação}

\begin{frame}{Identificação}
    \begin{itemize}
        \justifying
        \item É o processo de \textbf{antever} os possíveis riscos ao projeto (o levantamento de riscos deve ser feito com senso).
        \item É nesta etapa que os possíveis riscos e seus tipos são previstos.
    \end{itemize}
    \centering
    \includegraphics[width=\textwidth]{imagens/figura02.png}

\subsubsection{Análise}

\begin{frame}{Análise}
    \begin{itemize}
        \justifying
        \item É o processo de \textbf{classificar} os riscos identificados.
        \item O Gerente de Projeto, junto com a equipe, podem criar uma escala da gravidade do efeito dos riscos (exemplo: insignificante, tolerável, grave, catastrófico).
        \item Os riscos podem ser categorizados também por sua probabilidade de vir acontecer.
    \end{itemize}
\end{frame}

        \justifying
        \item É o processo de se precaver em relação aos riscos levantados.
        \item Cada risco deve ser estudado e ter pronto um plano para reação.

            \justifying
            \item Prevenção;
            \item Minimização;
            \item Contingência.
        \end{enumerate}
    \end{itemize}

\section{Atividade}

\begin{frame}{Atividade}
    \begin{itemize}
        \justifying
        \item Reúnam-se em grupos de \textbf{até 7 pessoas}.
        \item Definam um projeto de software sobre o qual vamos praticar o gerenciamento durante o semestre.

                \item Comecem a documentar a \textbf{seleção} e \textbf{organização} do grupo. Ex.: definindo os cargos/papéis de cada membro (incluindo o(a) líder/gerente), listando as razões da escolha de cada um para seu respectivo cargo.
