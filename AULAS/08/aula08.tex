\begin{frame}{Construção de Sistemas}
    \begin{itemize}
       \justifying
       \item<1-> É o \textbf{processo} de \textbf{criar um sistema executável completo} (chamado de \textit{build}), compilando e ligando os componentes do sistema, as bibliotecas externas, os arquivos de configuração e outras informações.

    \end{itemize}
\end{frame}

\begin{frame}{Construção de Sistemas}
    \begin{itemize}
        \justifying
        \item A construção é um processo complexo e potencialmente propenso a erros, já que três plataformas de sistema diferentes podem estar envolvidas:
    \end{itemize}
    \centering
    \includegraphics[width=0.9\textwidth]{imagens/figura02.png}
\end{frame}

\begin{frame}{Construção de Sistemas}
    \begin{itemize}
        \justifying
        \item Os \textbf{métodos ágeis} recomendam que a construção do sistema seja realizada com bastante frequência, com testes automatizados utilizados para descobrir problemas de software. As construções frequentes fazem parte de um processo de \textbf{integra-ção contínua}.
    \end{itemize}
    \centering
    \includegraphics[width=0.8\textwidth]{imagens/figura03.png}
\end{frame}

\begin{frame}{Construção de Sistemas}
    \begin{itemize}
        \justifying
        \item Nem sempre é possível implementar a integração contínua:
            \begin{enumerate}
                \justifying
                \item Se o sistema for muito grande.
                \item Se a plataforma de desenvolvimento for diferente da plataforma de destino.
            \end{enumerate}
        \item Uma solução nesses casos pode ser a \textbf{construção frequente}. Por exemplo, construção diária.
    \end{itemize}
\end{frame}

\subsection{Construção frequente}

\begin{frame}{Construção frequente}
    \begin{itemize}
        \justifying
        \item Vantagens
            \begin{itemize}
                \justifying
                \item Maiores chances de \textbf{encontrar problemas} oriundos das interações de componentes \textbf{logo no início do processo}.
                \item \textbf{Incentiva testes unitários} completos dos componentes.
                \item \textbf{Psicológica}
                    \begin{itemize}
                        \justifying
                        \item<2-> Os \textbf{desenvolvedores} são colocados \textbf{sob pressão} para não quebrar o \textit{build}, ou seja, eles tentam evitar o check-in de versões dos componentes que provoquem a falha do sistema inteiro.
                        \item<3> Portanto, eles \textbf{relutam em entregar versões} de novos componentes \textbf{que não foram adequadamente testadas}. Como conse-quência, gasta-se menos tempo durante o teste do sistema descobrindo e lidando com defeitos de software, que poderiam ter sido encontrados pelo desenvolvedor.
                    \end{itemize}
            \end{itemize}
    \end{itemize}
\end{frame}

\section{Gerenciamento de mudanças}

\begin{frame}{Gerenciamento de mudanças}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \justifying
                \footnotesize

\subsection{Ferramentas para apoiar o gerenciamento de mudanças}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{itemize}
        \justifying
        \item Podem ser tanto \alert<2>{\textbf{sistemas relativamente simples de rastreamento de problemas ou defeitos}} quanto um \alert<3>{\textbf{software integrado a um pacote de gerenciamento de configuração de sistemas de larga escala}}.
            \begin{itemize}
                \justifying
                \item<2> \href{https://docs.github.com/en/issues/tracking-your-work-with-issues/learning-about-issues/about-issues}{\textcolor{blue}{\underline{Github Issues}}}.
                \item<3> \href{https://www.ibm.com/br-pt/products/rational-clearcase}{\textcolor{blue}{\underline{IBM Rational ClearCase}}}.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \justifying
                \footnotesize
                \item Os sistemas de rastreamento de problemas permitem que qualquer pessoa relate um defeito ou faça uma sugestão de mudança no sistema, e controlam como o time de desenvolvimento respondeu a essas questões
            \end{itemize}
        \end{column}
        
        \begin{column}{0.6\textwidth}
            \centering
            \includegraphics[width=0.95\textwidth]{imagens/figura04_02.png}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \justifying
                \footnotesize
                \item Depois de uma solicitação de mudança ter sido enviada. ela é conferida para garantir que seja válida.
                \item A solicitação de mudança pode ser rejeitada nesse estágio
                    \begin{itemize}
                        \justifying
                        \tiny
                        \item O \textbf{defeito} pode já ter sido relatado e \textbf{corrigido}.
                        \item O que as pessoas \textbf{acreditam ser problemas} pode ser, na verdade, um \textbf{mal-entendido} quanto ao que se espera que o sistema faça.
                        \item As pessoas \textbf{pedem características} que \textbf{já foram implementadas}, \textbf{mas} sobre as quais elas \textbf{ainda não sabiam}.
                    \end{itemize}
            \end{itemize}
        \end{column}
        
        \begin{column}{0.6\textwidth}
            \centering
            \includegraphics[width=0.95\textwidth]{imagens/figura04_03.png}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \justifying
                \footnotesize
                \item Para as solicitações de mudança \textbf{válidas}, a próxima etapa do processo é a \textbf{avaliação da mudança} e o \textbf{cálculo dos custos}.
            \end{itemize}
        \end{column}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \justifying
                \footnotesize
                \item Um grupo à parte decide se é rentável para a empresa fazer a mudança no software.
                \item Nesse momento é considerado o impacto da mudança de um ponto de vista estratégico e organizacional, em vez de técnico.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.6\textwidth}
            \centering
            \includegraphics[width=0.95\textwidth]{imagens/figura04_05.png}

\begin{frame}{Ferramentas para apoiar o gerenciamento de mudanças}
    \begin{itemize}
        \justifying
        \item<1> Os fatores que influenciam a decisão sobre a implementação, ou não, de uma mudança incluem:
            \begin{itemize}
                \justifying
                \item<1> As consequências de não fazer a mudança.
                \item<1> Os benefícios da mudança.
                \item<1> O número de usuários afetados pela mudança.
                \item<1> Os custos de fazer a mudança.
                \item<1> O ciclo de lançamento do produto.
            \end{itemize}
        \item<2> Conforme o time de desenvolvimento altera os componentes de software, ele deve manter um \textbf{registro das alterações} feitas em cada componente. Ex.: \textbf{changelog}.
    \end{itemize}
\end{frame}
