\subsection{Planos de Projeto}

\begin{frame}{Planos de Projeto}
    \begin{itemize}
        \justifying
        \item Um plano de projeto estabelece os \textbf{recursos disponíveis} para o projeto. a \textbf{divisão de trabalho} e um \textbf{cronograma} para executar o trabalho.
        \item Também, o plano deve \textbf{identificar} a \textbf{abordagem} adotada para o \textbf{gerenciamento de risco}, bem como os \textbf{riscos para o projeto} e o \textbf{software} em desenvolvimento.
    \end{itemize}
\end{frame}

\begin{frame}{Planos de Projeto}
    \begin{itemize}
        \justifying
        \item Os planos costumam incluir as seguintes seções:
            \begin{enumerate}
                \item \alert<2>{Introdução}
                    \begin{itemize}
                        \justifying
                        \item<2> \textbf{Descreve} resumidamente os \textbf{objetivos} do projeto e \textbf{estabelece} as \textbf{restrições} (por exemplo, orçamento e tempo) que afetam o gerenciamento do projeto.
                    \end{itemize}
                \item \alert<3>{Organização do projeto}
                    \begin{itemize}
                        \justifying
                        \item<3> \textbf{Descreve} a maneira como o time de desenvolvimento é organizado, \textbf{as pessoas envolvidas} e \textbf{seus papéis no time}.
                    \end{itemize}
                \item Análise de risco
                \item \alert<4>{Requisitos de recursos de hardware e software}
                    \begin{itemize}
                        \justifying
                        \item<4> \textbf{Especifica} o \textbf{hardware} e o \textbf{software} de apoio \textbf{necessários} para realizar o desenvolvimento. Se houver necessidade de adquirir hardware, as estimativas de preço e de cronograma de entrega podem ser incluídas.
                    \end{itemize}
                \item Divisão de trabalho.
                \item Cronograma de projeto.
                \item Mecanismos de monitoramento e de divulgação.
            \end{enumerate}
    \end{itemize}
\end{frame}

\begin{frame}{Planos de Projeto}
    \begin{itemize}
        \justifying
        \item Os planos costumam incluir as seguintes seções:
        \begin{enumerate}
            \item Introdução
            \item Organização do projeto
            \item Análise de risco
            \item Requisitos de recursos de hardware e software
            \item \alert<1>{Divisão de trabalho}
                \begin{itemize}
                    \justifying
                    \item<1> \textbf{Estabelece a divisão} do projeto \textbf{em atividades} e \textbf{identifica} as \textbf{entradas} e as \textbf{saídas} \textbf{de cada atividade} de projeto.
                \end{itemize}
            \item \alert<2>{Cronograma de projeto}
                \begin{itemize}
                    \justifying
                    \item<2> \textbf{Mostra} as \textbf{dependências entre as atividades}, o \textbf{tempo} estimado necessário \textbf{para alcançar cada marco} e a \textbf{alocação de pessoas} às atividades.
                \end{itemize}
            \item \alert<3>{Mecanismos de monitoramento e de divulgação}
                \begin{itemize}
                    \justifying
                    \item<3> \textbf{Define} os \textbf{relatórios} de gestão \textbf{que devem ser produzidos}, \textbf{quando} devem ser produzidos e os \textbf{mecanismos de monitoramento} de projeto a serem utilizados.
                \end{itemize}
        \end{enumerate}
    \end{itemize}
\end{frame}

\begin{frame}{Planos de Projeto}
    \begin{itemize}
        \justifying
        \item Também é possível desenvolver uma série de planos suplementares.
    \end{itemize}

\subsection{O Processo de Planejamento}

\begin{frame}{O Processo de Planejamento}
    \begin{itemize}
        \justifying
        \item O planejamento de projeto é um \textbf{processo iterativo} que \textbf{começa} na \textbf{criação de um plano de projeto inicial} durante a fase inicial do projeto.
        \item À medida que \textbf{mais informações} sobre o sistema e o time de projeto \textbf{são disponibilizadas}, é necessário \textbf{revisar o plano} regularmente para refletir as alterações.
    \end{itemize}
\end{frame}

\section{Definição de Cronograma de Projeto}

\setbeamertemplate{background canvas}{%
       \tikz[remember picture,overlay] \node[opacity=0.15,inner sep=0pt] at (current page.center){\includegraphics[width=\paperwidth,height=\paperheight]{imagens/backgrounds/gantt-chart.png}};
    }
    
\begin{frame}{Definição de Cronograma de Projeto}
    \begin{itemize}
        \justifying
        \item A definição do cronograma de projeto é o processo de \textbf{decidir} \textbf{como o trabalho} em um projeto \textbf{será organizado} em tarefas distintas, \textbf{quando} e \textbf{de que maneira} essas tarefas serão \textbf{executadas}.
        \item<2-> São \textbf{estimados} os \textbf{prazos} necessários para realizar cada tarefa e o \textbf{esforço} necessário, além de serem \textbf{indicadas} as \textbf{pessoas} que trabalharão nas tarefas que foram identificadas, e também os recursos de \textbf{hardware} e de \textbf{software} necessários para completar cada tarefa.
    \end{itemize}
\end{frame}

\begin{frame}{Definição de Cronograma de Projeto}
    \begin{itemize}
        \justifying
        \item E se algo der errado?
        \item<2> Uma boa regra geral é estimar como se nada desse errado e, em seguida, aumentar a estimativa para cobrir os problemas previstos e, se possível aumentar também para imprevistos.
    \end{itemize}
